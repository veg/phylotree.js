      var tree = new phylotree.phylotree(data, {type:'phyloxml'});

      let height = 900;
      let width = 900;

      tree.render("#tree_display", {
                   height:height, 
                   width:width,
                   'left-right-spacing': 'fit-to-size', 
                   'top-bottom-spacing': 'fit-to-size'
                 })

			rendered_tree.style_edges (function (element, data) {

					}

    	});

			rendered_tree.update();

    })

    hljs.initHighlightingOnLoad();
  });
